import math
from typing import Dict

from sbmlsim.data import DataSet, load_pkdb_dataframe
from sbmlsim.fit import FitMapping, FitData
from sbmlutils.console import console

from pkdb_models.models import apixaban
from pkdb_models.models.apixaban.experiments.base_experiment import (
    ApixabanSimulationExperiment,
)
from pkdb_models.models.apixaban.experiments.metadata import (
    Tissue, Route, Dosing, ApplicationForm, Health, Fasting, ApixabanMappingMetaData
)

from sbmlsim.plot import Axis, Figure
    # noqa: E402
from sbmlsim.simulation import Timecourse, TimecourseSim

from pkdb_models.models.apixaban.helpers import run_experiments


class Raghavan2009(ApixabanSimulationExperiment):
    """Simulation experiment of Raghavan2009."""
    # FIXME: not tablet, but solution

    markers = {
        "group1": "s",
        "group2": "o",
    }
    groups = list(markers.keys())
    dose = 20 #[mg]

    infos_pk = {
        "[Cve_api]":{
            "measurements": ["apixaban", "apixaban2"],
            "kp": 0,
            "color": "black"
        },
        "[Cve_m1]": {
            "measurements": ["m1"],
            "kp": 1,
            "color": "black"
        },
        "Aurine_api": {
            "measurements": ["apixaban_urine"],
            "kp": 2,
            "color": "black"
        },
        "Aurine_m1": {
            "measurements": ["m1_urine"],
            "kp": 3,
            "color": "black"
        },
        "Aurine_m7": {
            "measurements": ["m7_urine"],
            "kp": 3,
            "color": "grey"
        },
        "Afeces_api": {
            "measurements": ["apixaban_feces"],
            "kp": 4,
            "color": "black"
        },
        "Afeces_m1": {
            "measurements": ["m1_feces"],
            "kp": 5,
            "color": "black"
        },
        "Afeces_m2": {
            "measurements": ["m2_feces"],
            "kp": 5,
            "color": "#1f77b4"
        },
        "Afeces_m7": {
            "measurements": ["m7_feces"],
            "kp": 5,
            "color": "grey"
        }
    }

    linestyle = {
        "apixaban": "solid",
        "apixaban2": "dashed",
        "m1": "dashed",
    }

    def datasets(self) -> Dict[str, DataSet]:
        dsets = {}
        for fig_id in ["Fig1", "Tab3", "Tab4"]:
            df = load_pkdb_dataframe(f"{self.sid}_{fig_id}", data_path=self.data_path)
            for label, df_label in df.groupby("label"):
                dset = DataSet.from_df(df_label, self.ureg)
                if "apixaban" in label:
                    Mr = self.Mr.api
                elif "m1" in label:
                    Mr = self.Mr.m1
                elif "m2" in label:
                    Mr = self.Mr.m2
                elif "m7" in label:
                    Mr = self.Mr.m7
                if "total" not in label and fig_id not in ["Tab3", "Tab4"]:
                    dset.unit_conversion("mean", 1 / Mr)
                if fig_id == "Tab4":
                    dset.unit_conversion("mean", 1 / self.Q_(100, "percent") * self.Q_(self.dose, "mg") / Mr)
                if "total" not in label:
                    dsets[label] = dset

        # console.print("datasets:", list(dsets.keys()))
        return dsets


    def simulations(self) -> Dict[str, TimecourseSim]:
        Q_ = self.Q_
        tcsims: Dict[str, TimecourseSim] = {}
        tcsims["control"] = TimecourseSim([
            Timecourse(
                start=0,
                end=301 * 60,  # [min]
                steps=500,
                changes={
                    **self.default_changes(),
                    "SOLDOSE_api": Q_(self.dose, "mg"),
                },
            )
        ])
        return tcsims

    # Fit Mappings
    def fit_mappings(self) -> Dict[str, FitMapping]:
        mappings: Dict[str, FitMapping] = {}
        infos = {
            **self.infos_pk,
        }
        for group in self.groups:
            for sid, info in infos.items():
                for name in info["measurements"]:
                    if "urine" in name:
                        tissue = Tissue.URINE
                    elif "feces" in name:
                        tissue = Tissue.FECES
                    else:
                        tissue = Tissue.PLASMA
                    mappings[f"fm_{name}_{group}"] = FitMapping(
                        self,
                        reference=FitData(
                            self,
                            dataset=f"{name}_{group}",
                            xid="time",
                            yid="mean",
                            count="count",
                        ),
                        observable=FitData(
                            self,
                            task="task_control",
                            xid="time",
                            yid=sid,
                        ),
                        metadata=ApixabanMappingMetaData(
                            tissue=tissue,
                            route=Route.PO,
                            application_form=ApplicationForm.SOLUTION,
                            dosing=Dosing.SINGLE,
                            health=Health.HEALTHY,
                            fasting=Fasting.FASTED,
                        ),
                    )

        return mappings


    def figures(self) -> Dict[str, Figure]:
        return {
            **self.figure_pk(),
        }

    def figure_pk(self) -> Dict[str, Figure]:
        fig = Figure(
            experiment=self,
            sid="PK",
            name=f"{self.__class__.__name__}",
            num_cols=2,
            num_rows=3,
            height=self.panel_height * 3 * 1.1,
            width=self.panel_width * 2,
        )
        plots = fig.create_plots(
            xaxis=Axis(self.label_time, unit=self.unit_time),
            legend=True,
        )
        for sid, info in self.infos_pk.items():
            kp = info["kp"]
            if "urine" in sid and "m1" in sid:
                label = "metabolites urine"
            elif "feces" in sid and "m1" in sid:
                label = "metabolites feces"
            elif "urine" in sid and ("m1" not in sid and "api" not in sid):
                continue
            elif "feces" in sid and ("m1" not in sid and "api" not in sid):
                continue
            else:
                label = self.labels[sid]
            plots[kp].set_yaxis(label, unit=self.units[sid])
            if sid.startswith("[Cve"):
                plots[kp].xaxis.min = -1
                plots[kp].xaxis.max = 81

        for group in self.groups:
            for sid, info in self.infos_pk.items():
                names = info["measurements"]
                kp = info["kp"]
                if group == "group1":
                    # Simulation
                    plots[kp].add_data(
                        task="task_control",
                        xid="time",
                        yid=sid,
                        label=f"sim: {self.dose}mg SOL" if "api" in sid or ("m1" in sid and kp == 1) else f"{sid.split("_")[1]} sim: {self.dose}mg SOL",
                        color=info["color"],
                    )
                for name in names:
                    if sid.startswith("[Cve"):
                        label = f"exp{group[-1]}: {self.dose}mg SOL [ng/ml]" if name == "apixaban" else f"exp{group[-1]}: {self.dose}mg SOL [µM]"
                    else:
                        label = f"exp{group[-1]}: {self.dose}mg SOL" if "api" in sid else f"{sid.split("_")[1]} exp{group[-1]}: {self.dose}mg SOL"
                    # Data
                    plots[kp].add_data(
                        dataset=f"{name}_{group}",
                        xid="time",
                        yid="mean",
                        count="count",
                        label=label,
                        color=info["color"],
                        linestyle=self.linestyle[name] if "[Cve" in sid else "",
                        marker=self.markers[group],
                        markeredgecolor=info["color"],
                    )

        return {fig.sid: fig}
